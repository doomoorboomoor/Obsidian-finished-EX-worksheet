> &=\int_{0}^{2}(x-x+2-x+4) d x+\int_{2}^{4}(x+x-2-x+4) d x \\
> &=\int_{0}^{2}(-x+2) d x+\int_{2}^{4}(x+2) d x
> \end{aligned}
